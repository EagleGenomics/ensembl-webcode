00224 sub associate_nodes {
00225     my ( $self, $nodes ) = @_;
00226     # sort by left index
00227     my @nodes     = sort { $a->{left_index} <=> $b->{left_index} } @$nodes;
00228     my $root      = $nodes[0];
00229     my $last_node = undef;
00230     # walk through them finding parents which enclose them
00231     # reset parent/child relationships
00232     for my $node (@nodes) {
00233         $node->{parent_id} = 0;
00234         $node->{parent}    = undef;
00235         $node->{children}  = [];

00237             my $parent = $self->_get_parent_by_index( $node, $last_node );
00238             if ( defined $parent ) {
00239                 $node->{parent_id} = $parent->taxon_id();
00240                 $node->{parent}    = $parent;
00241                 push @{ $parent->{children} }, $node;
00242             }
00243         }
00244         $node->{root_id} = $root->taxon_id();
00245         $node->{root}    = $root;

00284 sub collapse_tree {
00285     my ( $self, $root, $check ) = @_;
00286     my @nodes;
00287     $check ||= sub {
00288         my ($node) = @_;
00289         return (    $node->taxon_id() eq $root->taxon_id()
00290                  || ($node->dba() && scalar(@{$node->dba()}) > 0)
00291                  || scalar(@{$node->children()})!=1
00292                  || scalar( @{ $node->parent()->children() })>1 );
00293     };
00294     $root->traverse_tree(
00295         sub {
00296             my ( $node, $depth ) = @_;
00297             if ( $check->($node) ) {
00298                 push @nodes, $node;
00299             }
00300             return;
00301         } );

00306 sub _build_node_array {
00307     my ( $self, $leaf_nodes ) = @_;
00308     my $array;
00309     my $lmin = 0;
00310     my $rmax = 0;
00311     for my $node (@$leaf_nodes) {
00312         if ( !$node->{left_index} || !$node->{right_index} ) {
00313             print Dumper($node);
00314             throw "Indices not set for " . $node->to_string();
00315         }
00316         $array->[ $node->{left_index} ]  = 1;
00317         $array->[ $node->{right_index} ] = 1;
00318         $lmin =
00319           ( $lmin == 0 || $node->{left_index} < $lmin )
00320           ? $node->{left_index}
00321           : $lmin;
00322         $rmax = ( $rmax < $node->{right_index} ) ? $node->{right_index} : $rmax;
00323     }
00324     return [ $array, $lmin, $rmax ];
00325 }

00327 sub _restrict_set {
00328     my ( $self, $leaf_nodes, $input_set ) = @_;
00329     # build sparse array
00330     my ( $node_array, $lmin, $rmax ) =
00331       @{ $self->_build_node_array($leaf_nodes) };
00332     # hash leaf nodes by taxon_id to allow replacement
00333     my %leaves = map { $_->taxon_id() => $_ } @$leaf_nodes;
00334     my @output_set;
00335     # iterate over everything in the min/max range
00336     for my $node (
00337         grep {
00338                  $_->{right_index} >= $lmin
00339               || $_->{left_index} <= $rmax
00340         } @$input_set )
00341     {
00342         my $leaf = $leaves{ $node->taxon_id() };
00343         if ( defined $leaf ) {
00344             push @output_set, $leaf;
00345         } else {
00346             for ( my $i = $node->{left_index};
00347                   $i <= $node->{right_index}; $i++ )
00348             {
00349                 if ( $node_array->[$i] ) {
00350                     push @output_set, $node;
00351                     last;
00352                 }
00353             }
00354         }
00355     }
00356     return \@output_set;

00403 The TaxonomyNodeAdaptor handles TaxonomyNode objects, which represent nodes in the NCBI taxonomy
00404 
00405 There are two main practical uses for the :
00406 1. Accessing the taxonomy database to find species of interest and to then find those within the ENA databases e.g. to find all 
00407 descendants of the node representing Escherichia coli:
00408     my $node = $node_adaptor->fetch_by_taxon_id(562);
00409     for my $child (@{$node_adaptor->fetch_descendants($node)}) {
00410         my $dbas = $helper->get_by_taxon_id($node->taxon_id())
00411         if(defined $dbas) {
00412             for my $dba (@{$dbas}) {
00413                 # do something with the $dba
00414             }
00415         }
00416     }
00417 
00418 2. Placing ENA species into a taxonomic hierarchy and manipulating that hierarchy (mainly for use in the web interface)
00419     # iterate over all dbas from a registry or ENARegistryHelper
00420     for my $dba (@{$helper->get_all_DBAdaptors()}) {
00421         my $node = $node_adaptor->fetch_by_coredbadaptor($dba);
00422         # add the node to a hash set of leaf nodes if not already there
00423         if ( $leaf_nodes->{ $node->taxon_id() } ) {
00424             $node = $leaf_nodes->{ $node->taxon_id() };
00425         } else {
00426             $leaf_nodes->{ $node->taxon_id() } = $node;
00427         }
