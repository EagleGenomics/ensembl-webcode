00148 sub add_child {
00149     my ( $self, $child ) = @_;
00150     if ( !grep { $_->taxon_id() == $child->taxon_id() } @{ $self->{children} } )

00157 sub traverse_tree {
00158     my ( $self, $closure, $depth ) = @_;
00159     $depth ||= 0;
00160     $closure->( $self, $depth );
00161     for my $child ( @{ $self->children() } ) {
00162         $child->traverse_tree( $closure, $depth + 1 );
00163     }
00164     return;
00165 }

00167 sub count_leaves {
00168     my ($self,$sub) = @_;
00169     $sub ||= sub {
00170         my ($node) = @_;
00171         return ($node->is_leaf() || (defined $node->dba() && scalar(@{$node->dba()})>0));
00172     };
00173     my $n = 0;
00174     $self->traverse_tree(
00175         sub {
00176             my ( $node, $depth ) = @_;
00177             $n += $sub->($node)?1:0; 
00178             return;
00179         } );
00180     return $n;
00181 }

00193         sub {
00194             my ( $node, $depth ) = @_;
00195             $str .= ( "." x $depth ) . $node->to_string() . "\n";
00196             return;
00197         } );
